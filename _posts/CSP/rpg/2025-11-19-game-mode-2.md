<script>
  class StatisticsManager {
    constructor() {
      this.storageKey = 'rpgModeStatistics';
      this.stats = this.loadStats();
      this.chart = null;
    }

    loadStats() {
      const defaultStats = {
        chill: 0,
        action: 0,
        total: 0,
        history: []
      };

      try {
        const stored = localStorage.getItem(this.storageKey);
        return stored ? JSON.parse(stored) : defaultStats;
      } catch (e) {
        console.error('Failed to load statistics:', e);
        return defaultStats;
      }
    }

    saveStats() {
      try {
        localStorage.setItem(this.storageKey, JSON.stringify(this.stats));
      } catch (e) {
        console.error('Failed to save statistics:', e);
      }
    }

    recordSelection(mode) {
      const userId = this.getUserId();
      const timestamp = new Date().toISOString();
      
      this.stats.history.push({
        userId,
        mode,
        timestamp
      });

      if (mode === 'chill' || mode === 'action') {
        this.stats[mode]++;
        this.stats.total++;
      }

      this.saveStats();
      this.updateDisplay();
      
      console.log(`Recorded selection: ${mode} by user ${userId}`);
    }

    getUserId() {
      let userId = sessionStorage.getItem('rpgUserId');
      if (!userId) {
        userId = 'user_' + Math.random().toString(36).substr(2, 9);
        sessionStorage.setItem('rpgUserId', userId);
      }
      return userId;
    }

    getPercentage(mode) {
      if (this.stats.total === 0) return 0;
      return Math.round((this.stats[mode] / this.stats.total) * 100);
    }

    updateDisplay() {
      document.getElementById('chill-count').textContent = this.stats.chill;
      document.getElementById('action-count').textContent = this.stats.action;
      document.getElementById('total-count').textContent = this.stats.total;

      document.getElementById('chill-percentage').textContent = `${this.getPercentage('chill')}%`;
      document.getElementById('action-percentage').textContent = `${this.getPercentage('action')}%`;

      ['chill-count', 'action-count', 'total-count'].forEach(id => {
        const element = document.getElementById(id);
        element.classList.add('updated-stat');
        setTimeout(() => element.classList.remove('updated-stat'), 500);
      });

      this.updateChart();
    }

    updateChart() {
      if (!this.chart) return;

      this.chart.data.datasets[0].data = [this.stats.chill, this.stats.action];
      this.chart.update();
    }

    initChart() {
      const ctx = document.getElementById('stats-chart').getContext('2d');
      
      this.chart = new Chart(ctx, {
        type: 'doughnut',
        data: {
          labels: ['Chill / Cozy', 'Action'],
          datasets: [{
            data: [this.stats.chill, this.stats.action],
            backgroundColor: [
              'rgba(34, 197, 94, 0.7)',
              'rgba(239, 68, 68, 0.7)'
            ],
            borderColor: [
              'rgb(34, 197, 94)',
              'rgb(239, 68, 68)'
            ],
            borderWidth: 2,
            hoverOffset: 15
          }]
        },
        options: {
          responsive: true,
          maintainAspectRatio: false,
          plugins: {
            legend: {
              position: 'bottom',
              labels: {
                padding: 20,
                usePointStyle: true,
                font: {
                  size: 14
                },
              }
            },
            tooltip: {
              callbacks: {
                label: (context) => {
                  const label = context.label || '';
                  const value = context.raw || 0;
                  const percentage = this.getPercentage(context.dataIndex === 0 ? 'chill' : 'action');
                  return `${label}: ${value} votes (${percentage}%)`;
                }
              }
            }
          }
        }
      });
    }

    resetStats() {
      if (confirm('Are you sure you want to reset all statistics? This cannot be undone.')) {
        this.stats = {
          chill: 0,
          action: 0,
          total: 0,
          history: []
        };
        this.saveStats();
        this.updateDisplay();
        alert('Statistics have been reset.');
      }
    }
  }

  document.addEventListener('DOMContentLoaded', function() {
    const gameModeCards = document.querySelectorAll('.game-mode-card');
    const continueBtn = document.getElementById('continue-btn');
    const selectionHint = document.getElementById('selection-hint');
    const resetStatsBtn = document.getElementById('reset-stats');
    let selectedMode = null;

    const statsManager = new StatisticsManager();

    statsManager.initChart();
    statsManager.updateDisplay();

    if (!sessionStorage.getItem('rpgCreatorPreferences')) {
      sessionStorage.setItem('rpgCreatorPreferences', JSON.stringify({}));
    }

    gameModeCards.forEach(card => {
      card.addEventListener('click', function() {
        const mode = this.dataset.mode;
        
        gameModeCards.forEach(c => c.classList.remove('selected'));
        
        this.classList.add('selected');
        
        selectedMode = mode;
        
        continueBtn.disabled = false;
        
        selectionHint.textContent = `Selected: ${getModeDisplayName(mode)} mode`;
        selectionHint.className = 'text-black font-medium mt-4';
        
        const preferences = JSON.parse(sessionStorage.getItem('rpgCreatorPreferences'));
        preferences.gameMode = mode;
        sessionStorage.setItem('rpgCreatorPreferences', JSON.stringify(preferences));
        
        statsManager.recordSelection(mode);
        
        console.log('Game mode selected:', mode);
      });

      card.addEventListener('keydown', function(e) {
        if (e.key === 'Enter' || e.key === ' ') {
          e.preventDefault();
          this.click();
        }
      });

      card.setAttribute('tabindex', '0');
    });

    continueBtn.addEventListener('click', function() {
      if (selectedMode) {
        const preferences = JSON.parse(sessionStorage.getItem('rpgCreatorPreferences'));
        preferences.gameMode = selectedMode;
        sessionStorage.setItem('rpgCreatorPreferences', JSON.stringify(preferences));
        
        alert(`Proceeding with ${getModeDisplayName(selectedMode)} RPG creation!`);
      }
    });

    resetStatsBtn.addEventListener('click', function() {
      statsManager.resetStats();
    });

    function getModeDisplayName(mode) {
      const modeNames = {
        'chill': 'Chill / Cozy',
        'action': 'Action'
      };
      return modeNames[mode] || mode;
    }

    const preferences = JSON.parse(sessionStorage.getItem('rpgCreatorPreferences'));
    if (preferences.gameMode) {
      const previouslySelected = document.querySelector(`[data-mode="${preferences.gameMode}"]`);
      if (previouslySelected) {
        previouslySelected.click();
      }
    }
  });
</script>
